# Zip
* 두 개의 List의 값을 병렬적으로 추출함
```
alist = ['a1', 'a2', 'a3']
blist = ['b1', 'b2', 'b3']
for a, b in zip(alist, blist):
	print(a, b)
a1 b1
a2 b2
a3 b3

# 각 튜플끼리의 index끼리의 묶음
a, b, c = zip((1,2,3), (10,20,30), (100,200,300))
(1, 10, 100) (2, 20, 200) (3, 30, 300)

# 각 튜플의 같은 index끼리 묶어 합을 list로 변환
[sum(x) for x in zip((1,2,3), (10,20,30), (100,200,300))]
[111, 222, 333]
```
